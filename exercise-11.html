# Step 1
# Here I am replacing all "-999" values with na, the ~.x looks at all values in the data set and when it sees "-999" it replaces it with NA.

d <- replace_with_na_all(d, condition = ~.x == -999)

d <- d|>
  select(-"litter size", -"refs")

d$`maxlife(mo)` <- d$`max. life(mo)`

d$`weanmass(g)` <-d$`wean mass(g)`

# Here I am removing the old variable names by using the same - select process as the first line of code in this chunk.

d <- d|>
  select(-"max. life(mo)", -"wean mass(g)")
# Step 3
# Here I am using mutate function utilizing across which scans the entire data set. I then define that any numeric value is logged by utilizing where(is.numeric) which scans the data set and the condition I add at the end is log which gets the log values for all numeric values.

d <- d |>
  mutate(across(where(is.numeric), log))
